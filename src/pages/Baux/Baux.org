import * as React from "react";
import MainLayout from "../../components/layouts/MainLayout";
import SubLayout from "../../components/layouts/SubLayout";
import TabMenu from "../../ui/TabMenu";
import Ongoing from "./Ongoing";

interface Props {
  uiStore?: any;
}
class Baux extends React.Component<Props> {
  public state = {
    activeItemIndex: 0,
  };

  private setActiveItemIndex = (index: number) => {
    this.setState({ activeItemIndex: index });
  };

  public render() {
    return (
      <MainLayout>
        <SubLayout>
          <React.Fragment>
            <TabMenu
              activeItemIndex={this.state.activeItemIndex}
              onChange={this.setActiveItemIndex}
              items={[
                <span
                  style={{
                    display: "flex",
                    alignItems: "center",
                  }}
                >
                  En cours (5)
                </span>,
                <span
                  style={{
                    display: "flex",
                    alignItems: "center",
                  }}
                >
                  Archives (0)
                </span>,
              ]}
            />
            {this.state.activeItemIndex === 0 && <Ongoing />}
          </React.Fragment>
        </SubLayout>
      </MainLayout>
    );
  }
}

export default Baux;
